/* Dark & Light Mode Styles */ body.light-mode { --bg-color: #fff; --text-color: #000; --main-color: #007f73; } body.dark-mode { --bg-color: #000; --text-color: #fff; --main-color: #ff5722; } /* Modal Styling */ .modal { display: none; position: fixed; top: 50%; left: 50%; transform: translate(-50%, -50%); background: var(--bg-color); color: var(--text-color); padding: 2rem; box-shadow: 0 0 15px var(--main-color); border-radius: 1rem; z-index: 1000; } .modal-content { text-align: center; } .close-btn { position: absolute; top: 1rem; right: 1rem; cursor: pointer; font-size: 2rem; color: var(--main-color); } // Script for Light/Dark Mode Toggle const modeToggle = document.getElementById('mode-toggle'); const body = document.body; modeToggle.addEventListener('click', () => { body.classList.toggle('dark-mode'); body.classList.toggle('light-mode'); }); // Script for Modal Popup const modal = document.getElementById('modal'); const openModal = document.getElementById('open-modal'); const closeModal = document.getElementById('close-modal'); openModal.addEventListener('click', () => { modal.style.display = 'block'; }); closeModal.addEventListener('click', () => { modal.style.display = 'none'; }); // Close modal on outside click window.addEventListener('click', (event) => { if (event.target === modal) { modal.style.display = 'none'; } });
